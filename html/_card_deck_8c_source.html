96void CardDeck_delete(CardDeck* deck) {
97 if (deck == NULL) return; // if deck doesn't exist, do nothing
98
99 // set current node of deck towards head node
100 // this lets us delete each card starting from the beginning
101 deck->current = deck->head;
102
103 while (deck->head->successor != NULL) { // if the node after the head isn't the tail, delete that node
104 CardDeck_deleteNext(deck);

106

108 if(deck->head != NULL) free(deck->head);

276Card CardDeck_removeAt(CardDeck* deck, int index, deckError* result) {
277 // we must first update the predecessor to point past the node to be deleted
278 CardNode* preNode = NULL;
279 CardNode* delNode = NULL;
280
281 if (index == 0) { // if index is at 0, set predecessor to head
282 if (deck == NULL || deck->head->successor == NULL) {
283 if (result) *result = illegalCard;
284 return INVALID_CARD;
285 }
286 else {
287 preNode = deck->head;
288 }
289 }
290 else { // else, set predecessor to the node prior to the deleted node
291 preNode = CardDeck_cardNodeAt(deck, index - 1, result);
292 if (preNode == NULL) {
293 return INVALID_CARD;
294 }
295 }
296 delNode = preNode->successor; // find the deleted node
297
298 if (delNode == NULL) { // check if a node existed after preNode
299 if (result) *result = illegalCard; // // index went out of bounds
300 return INVALID_CARD;
301 }
302
303 // if all checks are fine, begin removal
304
305 Card delCard = delNode->card;
306
307 preNode->successor = delNode->successor; // set predecessor's successor pointer to the deleted node's successor
308 free(delNode);
309 if (result) *result = ok;
310 return delCard;
311}

377void CardDeck_shuffle(CardDeck* deck) {
378 if (deck == NULL || deck->head->successor == NULL) return; // if deck is null or empty, don't shuffle
379
380 CardDeck* shuffledDeck = CardDeck_create(); // empty deck, to be populated with shuffled cards
381 if (shuffledDeck == NULL) {
382 printf("Error: failed to allocate memory for shuffle deck.\n");
383 return;
384 }
385 deckError err = ok;
386
387 int deckSize = CardDeck_count(deck);
388
389 // loop transferring a random card from source deck to shuffle deck
390 for (int i = deckSize; i > 0; i--) {
391 int r = rand() % i;
392 Card removedCard = CardDeck_removeAt(deck, r, &err); // Value of card removed from source deck at index r
393
394 if (err != ok || removedCard.rank == INVALID_RANK)
395 {
396 printf("Error: could not remove card at %d from deck.\n", r);
397 return;
398 }
399
400 err = CardDeck_insertToTop(shuffledDeck, removedCard); // insert random card to top of shuffle deck
401
402 if (err != ok) {
403 printf("Error: could not insert card at %d to shuffle deck.\n", r);
404 return;
405 }
406 }
407
408 deck->head->successor = shuffledDeck->head->successor; // transfer temporary shuffled deck to source deck
409 free(shuffledDeck->head); // free allocated shuffle deck head
410 free(shuffledDeck); // free the allocated temporary shuffled deck
411 shuffledDeck = NULL;
412}
