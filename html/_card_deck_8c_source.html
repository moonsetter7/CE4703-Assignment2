266Card CardDeck_removeAt(CardDeck* deck, int index, deckError* result) {
267 // we must first update the predecessor to point past the node to be deleted
268 CardNode* preNode = NULL;
269 CardNode* delNode = NULL;
270
271 if (index == 0) { // if index is at 0, set predecessor to head
272 if (deck == NULL || deck->head->successor == NULL) {
273 if (result) *result = illegalCard;
274 return INVALID_CARD;
275 }
276 else {
277 preNode = deck->head;
278 }
279 }
280 else { // else, set predecessor to the node prior to the deleted node
281 preNode = CardDeck_cardNodeAt(deck, index - 1, result);
282 if (preNode == NULL) {
283 return INVALID_CARD;
284 }
285 }
286 delNode = preNode->successor; // find the deleted node
287
288 if (delNode == NULL) { // check if a node existed after preNode
289 if (result) *result = illegalCard; // // index went out of bounds
290 return INVALID_CARD;
291 }
292
293 // if all checks are fine, begin removal
294
295 Card delCard = delNode->card;
296
297 preNode->successor = delNode->successor; // set predecessor's successor pointer to the deleted node's successor
298 free(delNode);
299 if (result) *result = ok;
300 return delCard;
301}

365void CardDeck_shuffle(CardDeck* deck) {
366 if (deck == NULL || deck->head->successor == NULL) return; // if deck is null or empty, don't shuffle
367
368 CardDeck* shuffledDeck = CardDeck_create(); // empty deck, to be populated with shuffled cards
369 if (shuffledDeck == NULL) {
370 printf("Error: failed to allocate memory for shuffle deck.\n");
371 return;
372 }
373 deckError err = ok;
374
375 int deckSize = CardDeck_count(deck);
376
377 // loop transferring a random card from source deck to shuffle deck
378 for (int i = deckSize; i > 0; i--) {
379 int r = rand() % i;
380 Card removedCard = CardDeck_removeAt(deck, r, &err); // Value of card removed from source deck at index r
381
382 if (err != ok || removedCard.rank == INVALID_RANK)
383 {
384 printf("Error: could not remove card at %d from deck.\n", r);
385 return;
386 }
387
388 err = CardDeck_insertToTop(shuffledDeck, removedCard); // insert random card to top of shuffle deck
389
390 if (err != ok) {
391 printf("Error: could not insert card at %d to shuffle deck.\n", r);
392 return;
393 }
394 }
395
396 deck->head->successor = shuffledDeck->head->successor; // transfer temporary shuffled deck to source deck
397 free(shuffledDeck->head); // free allocated shuffle deck head
398 free(shuffledDeck); // free the allocated temporary shuffled deck
399 shuffledDeck = NULL;
