85int CardDeck_findMatchInHand(CardDeck* hand, CardDeck* playedDeck) {
86 // look at the top card on the played deck
87 Card* target = CardDeck_seeTop(playedDeck);
88 if (target == NULL)
89 {
90 // nothing to match against
91 return -1;
92
93 }
94
95 CardNode* node = hand->head->successor;
96 if (node == NULL) {
97 return -1; // return -1 if hand is empty
98 }
99
100 int index = 0;
101
102 // we walk through the linked list using current + gotoNextCard
103 while (node != NULL)
104 {
105 Card currentCard = node->card;
106
107 // check for same suit or rank
108 if (currentCard.suit == target->suit || currentCard.rank == target->rank)
109 {
110 // first match found return its index
111 return index;
112 }
113 // move to next card
114 node = node->successor;
115 index++;
116 }
117
118 return -1;
119}

130void Game_recycleHidden(Game* game) {

142 }
143
144 // Transfer all played cards into the hidden deck

156 }
