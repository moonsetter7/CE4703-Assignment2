12/*
13* Deals 8 cards to each player.
14* Always takes cards from the hidden deck and alternates betweeen player 1 and 2
15* so cards 0,2,4,6 go to player 1
16* cards 1,3,5,7 go to player 2
17*
18* @param game Pointer of the game.
19* @return deckError value indicating the game deal error status.
20*/
21
22deckError Game_deal(Game* game)
23{
24 deckError err = ok;
25 // deals 16 cards total; 8 for each player
26 for (int i = 0; i < 16; i++)
27 {
28 // take the top card from the hidden deck
29 Card takenCard = CardDeck_useTop(game->hidden, &err);
30 if (err != ok)
31 {
32 // if we cant get a card return error
33 return err;
34 }
35
36 // even i -> player 1, odd i -> player 2
37 if (i % 2 == 0)
38 {
39 // put card on top of player 1s deck
40 err = CardDeck_insertToTop(game->p1, takenCard);
41 }
42 else
43 {
44 // put card on top of player 2s deck
45 err = CardDeck_insertToTop(game->p2, takenCard);
46 }
47
48 if (err != ok)
49 {
50 // something went wrong adding the card to a players hand
51 return err;
52 }
53 }
54 return ok;
55}

68/*
69* Finds a card matching the top card of a played deck
70* from the player's hand. Two cards match if both share
71* either the same rank or the same suit.
72*
73* @param hand Pointer of the player's hand.
74* @param playedDeck pointer of the deck containing the played
75* cards.
76* @return The match status:
77* <ul>
78* <li>The index of the matched card in the player's hand</li>
79* <li><i>-1</i> if no match is found</li>
80* </ul>
81*/
82int CardDeck_findMatchInHand(CardDeck* hand, CardDeck* playedDeck) {
83 // look at the top card on the played deck
84 Card* target = CardDeck_seeTop(playedDeck);
85 if (target == NULL)
86 {
87 // nothing to match against
88 return -1;
89
90 }
91
92 CardNode* node = hand->head->successor;
93 if (node == NULL) {
94 return -1; // return -1 if hand is empty
95 }
96
97 int index = 0;
98
99 // we walk through the linked list using current + gotoNextCard
100 while (node != NULL)
101 {
102 Card currentCard = node->card;
103
104 // check for same suit or rank
105 if (currentCard.suit == target->suit || currentCard.rank == target->rank)
106 {
107 // first match found return its index
108 return index;
109 }
110 // move to next card
111 node = node->successor;
112 index++;
113 }
114
115 return -1;
116}

175/*
176* Plays one turn of the game.
177* <em>Steps:</em>
178* <ol>
179* <li>Try to find a matching card in a player's hand.</li>
180* <li>If no match is found:</li>
181* <ul>
182* <li>Recycle the hidden deck if it's empty</li>
183* <li>Draw one card from the player's hand.</li>
184* </ul>
185* <li>If a match is found:</li>
186* <ul>
187* <li>Remove the matching card from the player's hand at the given index</li>
188* <li>Put that card into the played deck</li>
189* </ul>
190* </ol>
191*
192* @param game Pointer of the game.
193* @param player The player in turn.
194*/
195void Game_playTurn(Game* game, int player) {
